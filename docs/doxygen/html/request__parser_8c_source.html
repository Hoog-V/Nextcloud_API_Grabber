24/*
25 * In the API the month is represented in text.
26 * This means it has to be converted to a number to be used in the tm struct.
27 */
28int month_str_to_int(const char *month) {
29 const char months[12][3] = {"Jan", "Feb", "Mar", "Apr", "May", "Jun",
30 "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"};
31 for (int i = 0; i < 12; i++) {
32 if (!strcmp(month, months[i])) {
33 return i;
34 }
35 }
36 return -1;
37}

41struct tm formatted_string_to_tm(const char *formatted_string) {
42 struct tm result;
43 if (formatted_string != NULL || strlen(formatted_string) < 28) {
44 const char month[4] = {formatted_string[8], formatted_string[9], formatted_string[10], '\0'};
45 const int year = strtol(YEAR_PTR(formatted_string), NULL, DECIMAL_BASE);
46 result.tm_hour = strtol(HOUR_PTR(formatted_string), NULL, DECIMAL_BASE);
47 result.tm_sec = strtol(SECONDS_PTR(formatted_string), NULL, DECIMAL_BASE);
48 result.tm_min = strtol(MINUTES_PTR(formatted_string), NULL, DECIMAL_BASE);
49 result.tm_mday = strtol(DOM_PTR(formatted_string), NULL, DECIMAL_BASE);
50 result.tm_year = year - CLOCK_START_YEAR;
51 result.tm_mon = month_str_to_int(month);
52 }
53 return result;
54}

61char *find_char_in_string(char *string, const char character_to_find){
62 if(string == NULL)
63 return NULL;
64 do{
65 string++;

67 if(*string == '\0')

72int preparse_propfind_resp(char *resp_body, const enum req_prop_type_t dataType){
73 resp_body = strstr(resp_body, "<d:prop>");

75 return PARSING_ERROR;

80 req_memory_ptrs[i] = resp_body;
81 resp_body = find_char_in_string(resp_body, '<');
82 if(resp_body != NULL)
83 *(resp_body-1) = '\0';
84 else
85 return PARSING_ERROR;
86 }
87#else
88 resp_body = find_char_in_string(resp_body, '>');
89 resp_body = find_char_in_string(resp_body, '>');
90 req_memory_ptrs[dataType] = resp_body;
91 resp_body = find_char_in_string(resp_body, '<');
92 if(resp_body != NULL)
93 *(resp_body-1) = '\0';
94 else
95 return PARSING_ERROR;
96 strcpy(attributes[dataType], req_memory_ptrs[dataType]);
97#endif
98 return 0;
99}
